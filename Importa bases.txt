        DtPassagem         8
        NuHash             8
        DsSentido        $ 5
        NuPrefixo          8
        NuValidador        8

    FORMAT
        DtPassagem       DATETIME18.
        NuHash           BEST10.
        DsSentido        $CHAR5.
        NuPrefixo        BEST6.
        NuValidador      BEST10.
        CdLinha          $CHAR6. ;
    INFORMAT
        DtPassagem       DATETIME18.
        NuHash           BEST10.
        DsSentido        $CHAR5.
        NuPrefixo        BEST6.
        NuValidador      BEST10.
        CdLinha          $CHAR6. ;
    INFILE "&TB_BL"
        LRECL=32767
        ENCODING="UTF-8"
		FIRSTOBS=1 
        DLM='|'
        MISSOVER
        DSD ;

        DtPassagem       : ?? ANYDTDTM19.
